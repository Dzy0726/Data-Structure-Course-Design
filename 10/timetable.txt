int getmaxplace(int arr[])
{
    int place = 1,max = arr[0];
    for(int i = 1;i < N;i++)
    {
        if(arr[i] > max)
        {
            max = arr[i];
        }
    }
    int temp = max/10;
    while(temp != 0)
    {
        place++;
        temp /= 10;
    }
    return place;
}
